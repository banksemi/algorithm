from functools import lru_cache


def solution(n, weak, dist):
    availables = dist.copy()
    @lru_cache(maxsize=None)
    def get_removal_points(start_point, distance) -> int:
        remove_bitmask = 0
        
        over = start_point + distance >= n
        for i in range(0, len(weak)):
            point = weak[i]
            if start_point <= point and start_point+distance >= point:
                remove_bitmask |= (1 << i)
            if over:
                if point <= start_point+distance - n and point < start_point:
                    remove_bitmask |= (1 << i)
        return remove_bitmask

    result = None
    bitmask = 2 ** len(weak) - 1
    def dfs(count):
        nonlocal result, bitmask
        
        # 이미 달성한 최소값을 달성할 수 없는 경우 빠르게 스킵
        if result is not None and result < count:
            return
        
        # 더 이상 꺼낼 사용자가 없는 경우 스킵
        if not availables:
            return
        # 가장 멀리 가는거 꺼내기
        current_dist = availables.pop()
        for i in range(len(weak)):
            if (1 << i) & bitmask == 0:
                continue
            start_point = weak[i]
            removal_bitmask = get_removal_points(start_point, current_dist)
            if bitmask ^ removal_bitmask == bitmask:
                continue
            original_mask = bitmask
            bitmask ^= removal_bitmask
                
            if bitmask:
                dfs(count+1)
            else:
                if result is None or result > count:
                    result = count
                    
            bitmask ^= removal_bitmask
        availables.append(current_dist)
        
    dfs(1)
    if result is None:
        return -1
    return result
