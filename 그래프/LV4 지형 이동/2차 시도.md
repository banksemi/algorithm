"""

우선 순위 큐
최소  비용인 곳으로만 간다. (최소 비용은 건설 비용)
모든 노드를 연결하면 종료
"""
from queue import PriorityQueue

dirs = [(0, 1), (0, -1), (-1,0), (1, 0)]

def solution(land, height):
    map_height = len(land)
    map_width = len(land[0])
    
    def move_tile(y, x):
        for dir in dirs:
            new_y = y + dir[0]
            new_x = x + dir[1]
            if new_y < 0 or new_x < 0:
                continue
            if new_y >= map_height or new_x >= map_width:
                continue
            yield new_y, new_x
        
        
    max_node = map_height * map_width
    visited = [[0] * map_width for i in range(map_height)]
    
    queue = PriorityQueue()
    queue.put((0, 0, 0))
    answer = 0
    while not queue.empty():
        cost, y, x = queue.get()
        if visited[y][x] != 0:
            continue
        answer += cost
        visited[y][x] = 1
        
        for new_y, new_x in move_tile(y, x):
            new_cost = abs(land[new_y][new_x] - land[y][x])
            if new_cost <= height:
                new_cost = 0
            if visited[new_y][new_x] == 0:
                queue.put((new_cost, new_y, new_x))
        
    return answer
