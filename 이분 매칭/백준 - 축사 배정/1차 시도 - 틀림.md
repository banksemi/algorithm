visited = set()  
def dfs(i):  
    if i == n:  
        return 0 # 추가 없음  
    result = 0  
    for next_node in edges[i]:  
        if next_node in visited:  
            continue  
        # 비어있으면  
        visited.add(next_node) # 방문 취급  
        added = dfs(i+1) + 1  
        result = max(added, result)  
        visited.remove(next_node)  
    # 못들어갈 때는 어차피 더 탐색할 필요가 없음  
    return result
