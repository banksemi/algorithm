구현 과정에서 시간이 오래 걸렸고, 런타임 에러를 해결하지 못했다.

이전과 마찬가지로 런타임 에러는 재귀함수의 스택 제한으로 인해 발생했다.
- 코테에서 재귀 함수를 사용할 때 꼭 sys.setrecursionlimit을 잊지 말자.


```
import json
import sys
sys.setrecursionlimit(1000000)

class Node:
    def __init__(self):
        self.childs = {}
        self.answer = False
        self.count = None
    
    def get_child(self, char):
        if char not in self.childs:
            self.childs[char] = Node()
        return self.childs[char]
    
def solution(words):
    main_node = Node()
    
    def make_tree(node, word, i):
        if i == len(word):
            node.answer = True
            return
        make_tree(node.get_child(word[i]), word, i+1)
    
    for word in words:
        make_tree(main_node, word, 0)
    
    def make_word_count(node):
        count = 0
        if node.answer:
            count += 1
        for next_node in node.childs.values():
            count += make_word_count(next_node)
        node.count = count
        return count
    make_word_count(main_node)
    
    answer = 0
    def found(node, level):
        nonlocal answer
        if node.count == 1:
            answer += level
            return
        
        # 더 탐색해야함
        if node.answer:
            answer += level
            
        for next_node in node.childs.values():
            found(next_node, level+1)
            
    found(main_node, 0)
    return answer
    
```
