=트리를 미리 안만들고 정렬된 값을 하는 방법

2~5초의 실행 시간이 200ms까지 단축된다.

```
def solution(words):
    answer = 0
    words = sorted(words)
    def dfs(min_i, max_i, level):
        nonlocal answer
        
        if min_i + 1 == max_i:
            answer += level
            return
        
        groups = {}
        for i in range(min_i, max_i):
            word = words[i]
            if len(word) == level:
                answer += level
                continue
            char = word[level]
            
            if char not in groups:
                groups[char] = [i, i]
            groups[char][1] = i+1
        
        for char in groups:
            dfs(groups[char][0], groups[char][1], level + 1)

    dfs(0, len(words), 0)
    return answer
```
