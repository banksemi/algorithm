블럭을 시뮬레이션하는 문제이다. 

특이사항은 없지만, 블럭을 아래로 당기는 과정에서 잘못 구현하여 테스트 케이스를 통과하지 못했다.
- for new_y in range(height-1, -1, -1): # 틀린 부분
- for new_y in range(height-1, y, -1): # 정답

```
"""
4:30 실패 63.6점
4:32 통과
"""

def solution(m, n, board):
    board = [[i for i in row] for row in board]
    height = m
    width = n
    def get_remove_positions():
        result = set()
        for y in range(0, height-1):
            for x in range(width-1):
                temp = set()
                temp.add(board[y][x])
                temp.add(board[y][x+1])
                temp.add(board[y+1][x])
                temp.add(board[y+1][x+1])
                if len(temp) != 1:
                    continue
                item = temp.pop()
                if item is None:
                    continue
                for i in [(y,x),(y+1,x),(y,x+1),(y+1,x+1)]:
                    result.add(i)
        return result
        
    def remove(positions):
        for y, x in positions:
            board[y][x] = None
        ...
    def block_down():
        for y in range(height-2, -1, -1): # 아래부터 진행하기때문에 순차적으로 아래로 쌓임
            for x in range(width):
                for new_y in range(height-1, y, -1): # 틀린 부분
                    if board[new_y][x] is None:
                        board[new_y][x] = board[y][x]
                        board[y][x] = None
                        break
                        
    answer = 0
    while True:
        blocks = get_remove_positions()
        if len(blocks) == 0:
            return answer
        answer += len(blocks)
        remove(blocks)
        block_down()
```
