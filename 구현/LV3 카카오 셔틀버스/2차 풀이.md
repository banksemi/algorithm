"""
5:01 시작
5:16 종료

가장 마지막에 타는 인원을 구한다.
- 만약 해당 대기열(게을러서 가장 마지막에 도착)에 탈 수 없다면 1분 먼저 온다.

가장 마지막 버스를 탈 예정이다.
- 마지막 버스에 여유 공간이 있으면 (버스 출발 시간에 온다.)
- 마지막 버스에 여유 공간이 없으면
    - 해당 대기열에 가장 마지막에 도착했을 때 탈 수 있는지 확인한다.
    - 탈 수 없으면 1분 먼저 온다.
"""
from collections import deque, defaultdict
def time_to_int(text):
    arr = [int(i) for i in text.split(':')]
    return arr[0] * 60 + arr[1]

def time_to_str(value):
    h, m = divmod(value, 60)
    return f'{h:02}:{m:02}'

def solution(n, t, m, timetable):
    queue = deque()
    count_dict = defaultdict(int)

    for text in timetable:
        time = time_to_int(text)
        count_dict[time] += 1
    for time in sorted(count_dict.keys()):
        queue.append([time, count_dict[time]])
    bus_times = []
    last_time = 540 + t * (n-1)
    answer = 0
    for time in range(540, 540 + t * (n), t):
        available = m
        while available and queue and queue[0][0] <= time: # 버스 이전(혹은동일)시간에 온 사람들
            temp = min(available, queue[0][1])
            queue[0][1] -= temp
            available -= temp
            if time == last_time and available == 0:
                answer = queue[0][0]-1
        
            if queue[0][1] == 0:
                queue.popleft()
            
    if available:
        return time_to_str(last_time)
    return time_to_str(answer)
