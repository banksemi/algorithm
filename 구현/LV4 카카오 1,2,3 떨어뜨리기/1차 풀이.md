def get_number_array(number, n):
    arr = [1] * n
    i = 0
    for _ in range(number - n):
        arr[i] += 1
        if arr[i] == 3:
            i += 1
    return sorted(arr)

def solution(edges, target):
    nodes = {i: [] for i in range(1, len(edges) + 2)}
    nodes_index = {i: 0 for i in range(1, len(edges) + 2)}
    
    edges.sort()
    for i, j in edges:
        nodes[i].append(j)
    
    target_dict = {}
    for i, value in enumerate(target):
        target_dict[i+1] = value
        
    def next_game(i):
        if not nodes[i]:
            return i
        
        next_node = nodes[i][nodes_index[i]]
        result = next_game(next_node)
        nodes_index[i] = (nodes_index[i] + 1) % len(nodes[i]) 
        return result
    
    node_arrived = {i: [] for i in range(1, len(edges) + 2)}
    total_count = 0
    for i in range(0, 10000):
        node = next_game(1)
        node_arrived[node].append(i)
        total_count += 1
        status = 'finished'
        for j, arr in node_arrived.items():
            count = len(arr)
            if count > target_dict[j]:
                status = 'failed'
                break
            
            if count * 3 < target_dict[j]:
                status = 'continue'
                break
        if status == 'failed':
            return [-1]
        if status == 'finished':
            break
    
    answer = [0] * total_count
    for j, arr1 in node_arrived.items():
        count = len(arr1)
        target = target_dict[j]
        for k, g in enumerate(get_number_array(target, count)):
            answer[arr1[k]] = g
            
    return answer
