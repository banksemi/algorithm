부분 배열이 고유 변수들로만 이루어진 집합 개수 찾기

```
def solution(M, A):
    l = 0
    r = 0
    answer = 0
    ones = set()
    overflow = {}

    def add(value):
        if value not in ones:
            ones.add(value)
            return
        else:
            if value in overflow:
                overflow[value] += 1
            else:
                overflow[value] = 1

    def remove(value):
        if value in overflow:
            overflow[value] -= 1
            if overflow[value] == 0:
                del overflow[value]
        else:
            ones.remove(value)
    
    while l != len(A):
        # r을 늘릴 수 있으면 늘리기
        if r != len(A) and A[r] not in ones:
            answer += 1
            add(A[r])
            r += 1
            continue
        
        # r을 늘릴 수 없었던 경우는 l을 한칸 옮기기
        remove(A[l])
        answer += r-l-1
        l += 1

    return min(answer, 1_000_000_000)
```
