"""
N개의 배열이 있을 때 배열에 등장하지 않은 가장 작은 양수(1이상)을 찾아라.

N 개수: 10만
각 요소 범위: -100만~100만
"""
def solution(A):
    items = set(A)

    for i in range(1, 1_000_001):
        if i not in items:
            return i
