def solution(A):
    # Implement your solution here

    # - 부호 2개 or 0개

    A.sort()
    result = -1000 * 1000 * 1000
    
    # 음수 케이스 고려
    result = max(A[0] * A[1] * A[-1], result)

    # 양수 케이스
    result = max(A[-3] * A[-2] * A[-1], result)

    return result
