def parse(char):
    if char == 'A':
        return 1
    if char == 'C':
        return 2
    if char == 'G':
        return 3
    if char == 'T':
        return 4
    raise Exception("파싱 에러")


def solution(S, P, Q):
    answer = []
    counters = [[0] * 4 for i in S]
    for i in range(len(S)):
        if i > 0:
            for j in range(0, 4):
                counters[i][j] = counters[i-1][j]
        counters[i][parse(S[i]) - 1] += 1
    
    for p, q in zip(P, Q):
        for i in range(4):
            result = counters[q][i]
            if p > 0:
                result -= counters[p-1][i]
            if result > 0: # TODO
                answer.append(i+1)
                break
    return answer
