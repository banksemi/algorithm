"""
블럭을 나누었을 때 peak가 하나 이상 포함되도록 하는 최대 k 찾기

"""
def solution(A):
    N = len(A)

    peaks = set()
    for i in range(1, N-1):
        if A[i-1] < A[i] and A[i] > A[i+1]:
            peaks.add(i)
    

    
    available_k = set()
    for a in range(1, int(N ** 0.5) + 1):
        b, m = divmod(N, a)

        if m == 0:
            available_k |= set((a, b))
    

    def check(k):
        no_peak_group = set(range(0, k))

        for peak in peaks:
            group_id = peak // (N // k)
            if group_id in no_peak_group:
                no_peak_group.remove(group_id)
            if not no_peak_group:
                return True
        return False

    for k in sorted(available_k, reverse=True):
        if check(k):
            return k
    return 0
