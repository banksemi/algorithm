def solution(N):
    answer = None
    for a in range(1, int(N ** 0.5) + 1):
        b, m = divmod(N, a)
        if m == 0:
            perimeter = 2 * (a + b)
            if answer is None or perimeter < answer:
                answer = perimeter
    return answer
