```
"""
4:29
# 0을 뜯고 시작할때 
# 0을 뜯지 않고 시작할때 DP로 문제 해결
"""
def solution(sticker):
    answer = 0
    def run(sticker, use_first):
        if use_first and len(sticker) == 1: # 예외 처리
            return sticker[0]
        dp = {}
        dp[-1] = 0
        if use_first:
            dp[0] = sticker[0]
            sticker = sticker[0:-1]
        else:
            dp[0] = 0
        for i in range(1, len(sticker)):
            dp[i] = max(dp[i - 2] + sticker[i], dp[i-1])
        
        return dp[len(sticker)-1]
    answer1 = run(sticker, True)
    answer2 = run(sticker, False)
    return max(answer1, answer2)
```
