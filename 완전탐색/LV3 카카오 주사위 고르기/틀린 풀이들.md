```
"""
주사위를 반반 가져가려고한다.
주사위에 있는 눈금은 모두 다른데 승리 확률이 높아지는 주사위 조합을 가져가려고 한다.

결과 반환 규칙
- 1부터 오름차순 반환

N = 10

"""
from itertools import permutations, product
def solution(dice):
    # print(6 ** 10) ->60466176 완전 탐색 가능 약간이 최적화는 필요할 수 있음
    
    best_score = 0
    best_a = []
    for dice_indexes in permutations(range(0,len(dice))):
        score = 0
        for select_indexes in product(list(range(6)), repeat=len(dice)):
            a_sum = 0
            b_sum = 0
            for i, j in enumerate(select_indexes):
                if i < len(dice)//2:
                    a_sum += dice[dice_indexes[i]][j]
                else:
                    b_sum += dice[dice_indexes[i]][j]
            if a_sum > b_sum:
                score += 1
        if best_score < score:
            best_score = score
            best_a = dice_indexes[0:len(dice)//2]
        
    return [i+1 for i in best_a]
```

```
"""
주사위를 반반 가져가려고한다.
주사위에 있는 눈금은 모두 다른데 승리 확률이 높아지는 주사위 조합을 가져가려고 한다.

결과 반환 규칙
- 1부터 오름차순 반환

N = 10

"""
from itertools import combinations, product
def solution(dice):
    # print(6 ** 10) ->60466176 완전 탐색 가능 약간이 최적화는 필요할 수 있음
    
    best_score = 0
    best_a = []
    for a_indexes in combinations(range(0,len(dice)), len(dice)//2):
        b_indexes = set(range(0, len(dice))) - set(a_indexes)
        a_indexes = list(a_indexes)
        b_indexes = list(b_indexes)
        score = 0
        for select_indexes in product(list(range(6)), repeat=len(dice)):
            a_sum = 0
            b_sum = 0
            for i, j in enumerate(select_indexes):
                if i < len(dice)//2:
                    a_sum += dice[a_indexes[i]][j]
                else:
                    b_sum += dice[b_indexes[i-(len(dice)//2)]][j]
            if a_sum > b_sum:
                score += 1
        if best_score < score:
            best_score = score
            best_a = a_indexes
        
    return [i+1 for i in sorted(best_a)]
```

부분 최적화

```
"""
주사위를 반반 가져가려고한다.
주사위에 있는 눈금은 모두 다른데 승리 확률이 높아지는 주사위 조합을 가져가려고 한다.

결과 반환 규칙
- 1부터 오름차순 반환

N = 10

"""
from itertools import combinations, product
def solution(dice):
    # print(6 ** 10) ->60466176 완전 탐색 가능 약간이 최적화는 필요할 수 있음
    
    best_score = 0
    best_a = []
    skip = set()
    for a_indexes in combinations(range(0,len(dice)), len(dice)//2):
        b_indexes = set(range(0, len(dice))) - set(a_indexes)
        a_indexes = tuple(sorted(list(a_indexes)))
        b_indexes = tuple(sorted(list(b_indexes)))
        if a_indexes in skip:
            continue
        a_score = 0
        b_score = 0
        for select_indexes in product(list(range(6)), repeat=len(dice)):
            a_sum = 0
            b_sum = 0
            for i, j in enumerate(select_indexes):
                if i < len(dice)//2:
                    a_sum += dice[a_indexes[i]][j]
                else:
                    b_sum += dice[b_indexes[i-(len(dice)//2)]][j]
            if a_sum > b_sum:
                a_score += 1
            if a_sum < b_sum:
                b_score += 1
        if best_score < a_score:
            best_score = a_score
            best_a = a_indexes
        
        if best_score < b_score:
            best_score = b_score
            best_a = b_indexes
        skip.add(a_indexes)
        skip.add(b_indexes)
        
    return [i+1 for i in sorted(best_a)]
```
